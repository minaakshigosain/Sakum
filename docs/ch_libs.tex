\section{Library Functions}\label{ch:libs}

The \texttt{assembly/sakum\_lib\_*.s} files are topic-keyed library
routines cited by the update signals (SIMD / AVX / NEON / RVV / WASM /
quantum / memory-safety / crypto / overflow / bounds).  Each exposes a
\texttt{\_sakum\_lib\_<topic>} entry plus a self-test \texttt{\_main}.

\subsection{Inventory}

\begin{table}[H]
\centering
\begin{tabularx}{\textwidth}{lXl}
\toprule
File & Routine & Behaviour \\
\midrule

  horizontal-sum of $n$ floats \\

\texttt{sakum\_lib\_bounds.s} & \texttt{\_sakum\_lib\_bounds} & bounds-checked

\noindent \textbf{Accuracy flag:} several library files are duplicates ---
\texttt{vector} $\equiv$ \texttt{simd} $\equiv$ \texttt{rvv} (all AVX2
float-fold, despite the \texttt{rvv} label), and
\texttt{overflow} $\equiv$ \texttt{bounds} $\equiv$ \texttt{memory.safe}
(all bounds-checked reads).  The \texttt{rvv} file's name claims RISC-V
Vector but its body is x86 AVX2.

\subsection{Double-FNV crypto (no SHA)}

\begin{lstlisting}[language=SakumAsm,caption={Double-FNV-1a (sakum\_lib\_crypto.s:8).}]
    mov eax, 0x811C9DC5
    mov r8d, 0x1000193
    xor rcx, rcx
.cl: cmp rcx, rsi; jge .cd
    movzx edx, byte ptr [rdi+rcx]
    xor eax, edx; imul eax, eax, 16777619
    xor r8d, edx; imul r8d, r8d, 16777619
    inc rcx; jmp .cl
\end{lstlisting}

\subsection{Bounds-checked read}

\begin{lstlisting}[language=SakumAsm,caption={Bounds check (sakum\_lib\_bounds.s:6).}]
    cmp rdx, rsi
    jge .oob          # index >= length -> out of bounds
    mov rax, [rdi + rdx*8]
    ret
.oob:
    mov rax, -1
    ret
\end{lstlisting}
